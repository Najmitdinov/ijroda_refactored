🔐 Firestore Security Rules (nusxalab Firebase Console ga qo'ying)
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Users: faqat o'z profilini o'qiy/yozishi mumkin
    // Admin/superadmin barchani boshqara oladi
    match /users/{uid} {
      allow read: if request.auth != null;
      allow create: if request.auth.uid == uid;
      allow update: if request.auth.uid == uid
        || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role
           in ['admin', 'superadmin'];
      allow delete: if get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role
           == 'superadmin';
    }

    // Documents: o'z hujjatlarini CRUD qiladi
    // Admin/superadmin barchasini ko'ra oladi
    match /documents/{docId} {
      allow read: if request.auth != null
        && (resource.data.userId == request.auth.uid
          || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role
             in ['admin', 'superadmin']);
      allow create: if request.auth != null
        && request.resource.data.userId == request.auth.uid;
      allow update, delete: if request.auth != null
        && (resource.data.userId == request.auth.uid
          || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role
             in ['admin', 'superadmin']);
    }

    // AI Chat: faqat o'z chatlarini boshqaradi
    match /users/{uid}/chats/{chatId} {
      allow read, write: if request.auth.uid == uid;
      match /messages/{msgId} {
        allow read, write: if request.auth.uid == uid;
      }
    }

    // AI Usage: faqat o'z statsini yozadi, admin o'qiy oladi
    match /aiUsage/{docId} {
      allow create, update: if request.auth != null
        && request.resource.data.uid == request.auth.uid;
      allow read: if get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role
        in ['admin', 'superadmin'];
    }

    // Fishkalar: autentifikatsiya talab qilinadi
    match /fishkalar/{id} {
      allow read, write: if request.auth != null;
    }

    // Xodimlar, Sektorlar
    match /xodimlar/{id} { allow read, write: if request.auth != null; }
    match /sektorlar/{id} { allow read, write: if request.auth != null; }
    match /tashkilotlar/{id} { allow read, write: if request.auth != null; }
    match /muhim_topshiriqlar/{id} { allow read, write: if request.auth != null; }
    match /ai_hisobot_buyruqlar/{id} {
      allow read: if request.auth != null;
      allow write: if get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role
        in ['admin', 'superadmin'];
    }

    // AI shablonlar va avtomatik hujjat yaratish
    match /ai_document_templates/{id} {
      allow read, write: if request.auth != null
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role
          in ['org_admin', 'admin', 'superadmin'];
    }
    match /ai_knowledge_documents/{id} {
      allow read, write: if request.auth != null
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role
          in ['org_admin', 'admin', 'superadmin'];
    }
    match /ai_generated_documents/{id} {
      allow read, write: if request.auth != null;
    }
  }
}
Firebase Storage Rules (shablon fayllari uchun)
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /ai_templates/{uid}/{fileName} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
    }
    match /ai_knowledge/{uid}/{fileName} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
    }
  }
}
